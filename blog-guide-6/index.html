// context.js
class ThemeProvider extends React.Component {
    switchTheme = () => {
        const theme = this.state.theme === "dark" ? "light" : "dark"
        localStorage.setItem("theme", theme)
        // Thm dng ny
        document.body.className = theme
        this.setState({ theme })
    }
    componentDidMount() {
        const theme = localStorage.getItem("theme") || "dark"
        this.setState({ theme })
        // Thm dng ny
        document.body.className = theme
    }
